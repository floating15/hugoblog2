Summary of Data Cleaning Skills

2019/07/08

Summary

df.head(n)
df.sample(n)
df.info()
df.describe()
df.shape
df.values
df.columns
df.index

Sort

df.sort_values(
    by="col_name",
    axis=0,
    ascending=True,
    inplace=False
)

df.sort_index(
    level=["outter_index", "inner_index"],
    ascending=False
)

Set index

df.set_index("col")
df.ser_index("outter_index", "inner_index")

df.reset_index()
df.reset_index(drop=False)

Subset and Slice

# subset by row
df[df["col"] > 0]

df[
    (df.col1 > 0) &
    (df.col2 >0)
]

df[
    df.col.isin(["xxx", "yyy"])
]


# with loc method
df.loc[
    ["outer_index1", "outer_index_2"]
]

df.loc[
    [
        ("outer_index_1", "inner_index_1"),
        ("outer_index_2", "inner_index_2"),
        ("outer_index_3", "inner_index_3"),
    ]
]

df.loc[
    "outer_index_i" : "outer_index_j"
]

df.loc[
    ("outer_index_i", "inner_index_i"):
    ("outer_index_j", "inner_index_j")
]

# subset & Slice by col
df["col"]
df.col
df[df.col == df.col.max()].col  # equal to df.col.max()
df[
    ["col1", "col2"]
]

# with loc method
df_sort_id.loc[
    [
        ("outer_index_1", "inner_index_1"),
        ("outer_index_2", "inner_index_2"),
        ("outer_index_3", "inner_index_3"),
    ] ,
    ["col1",  "col2"]
]

df_sort_id.loc[
    ("Julia", "1"): ("Julia", "3") ,
    "col_i": "col_j"
]

Aggregate

df.col.mean()
df[["col1", "col2"]].mean()

def pct30(col):
    return col.quantile(0.3)

df.col.agg(min)
df["col1", "col2"].agg([min, max, np.mean, np.median, pct30])
df["col1", "col2"].agg([cumsum, cummin, cumprod])

df.mean(axis="columns") # calculate for every rows
df.mean(axis="index") # calculate for every columns

Duplicate Value

df.drop_duplicates(
    subset="col"
)

df.drop_duplicates(
    subset=["col1", "col2"]
)

Counting

df.col.value_counts(
    sort=True,
    normalize=False
)

# count missing value
df.isna()
df.isna().any()
df.isna().sum()
df.isna().sum().plot(kind="bar")

Groupby

# groupby method
df.groupby(
    "col"
)
df.groupby(
    ["col1", "col2"]
)

# pivot table
df.pivot_table(
    values=["col1", "col2"], 
    index=["col3", "col4"], 
    columns=["col5", "col6"],
    aggfunc='mean', 
    fill_value=None, 
    margins=False, 
    dropna=True, 
    margins_name='All'
)
